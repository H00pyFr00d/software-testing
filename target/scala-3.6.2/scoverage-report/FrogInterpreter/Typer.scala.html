57   def tyCheck(ctx: Env, e: Expr, ty: Type): Unit = (e, ty) match {
58     // BEGIN ANSWER

60     case (Lambda(x, e), TyFun(ty1, ty2)) =>

264

270

276     case First(e) => tyInfer(ctx, e) match {
277       case TyPair(ty1, _) => ty1

280     case Second(e) => tyInfer(ctx, e) match {
